# Express 测试与部署

代码经过测试验证后才能让人放心，配置完善的部署流程才能让应用稳定运行。本章将学习 Express 应用的测试策略和部署方案。

## 单元测试与集成测试

### 测试工具链

```bash
npm install --save-dev jest supertest
```

- **Jest**：JavaScript 测试框架，提供断言、Mock 和测试运行器
- **Supertest**：HTTP 断言库，可以在不启动真实服务器的情况下测试 Express 应用

### 基础测试用例

```js
// app.js
const express = require('express');
const app = express();

app.use(express.json());

app.get('/health', (req, res) => {
  res.json({ status: 'ok' });
});

app.post('/users', (req, res) => {
  const { name, email } = req.body;
  if (!name || !email) {
    return res.status(400).json({ error: '姓名和邮箱不能为空' });
  }
  res.status(201).json({ id: 1, name, email });
});

module.exports = app; // 导出供测试使用
```

```js
// __tests__/app.test.js
const request = require('supertest');
const app = require('../app');

describe('GET /health', () => {
  test('应返回健康状态', async () => {
    const response = await request(app)
      .get('/health')
      .expect(200);

    expect(response.body.status).toBe('ok');
  });
});

describe('POST /users', () => {
  test('应创建用户', async () => {
    const response = await request(app)
      .post('/users')
      .send({ name: 'Alice', email: 'alice@example.com' })
      .expect(201);

    expect(response.body).toMatchObject({
      id: expect.any(Number),
      name: 'Alice',
      email: 'alice@example.com'
    });
  });

  test('缺少字段时应返回 400', async () => {
    const response = await request(app)
      .post('/users')
      .send({ name: 'Alice' })
      .expect(400);

    expect(response.body.error).toBe('姓名和邮箱不能为空');
  });
});
```

### 内存数据库方案（MongoDB）

```bash
npm install --save-dev mongodb-memory-server
```

```js
// tests/setup.js
const { MongoMemoryServer } = require('mongodb-memory-server');
const mongoose = require('mongoose');

let mongod;

beforeAll(async () => {
  mongod = await MongoMemoryServer.create();
  const uri = mongod.getUri();
  await mongoose.connect(uri);
});

afterAll(async () => {
  await mongoose.disconnect();
  await mongod.stop();
});

afterEach(async () => {
  // 每个测试后清空数据
  const collections = mongoose.connection.collections;
  for (const key in collections) {
    await collections[key].deleteMany();
  }
});
```

### 认证路由测试

```js
describe('受保护路由', () => {
  test('无 Token 时应返回 401', async () => {
    await request(app)
      .get('/profile')
      .expect(401);
  });

  test('有效 Token 时应返回用户数据', async () => {
    // 先登录获取 Token
    const login = await request(app)
      .post('/login')
      .send({ email: 'alice@example.com', password: 'password' });

    const token = login.body.token;

    await request(app)
      .get('/profile')
      .set('Authorization', `Bearer ${token}`)
      .expect(200);
  });
});
```

### 优雅关闭

```js
const server = app.listen(port, () => {
  console.log(`服务运行在端口 ${port}`);
});

const gracefulShutdown = (signal) => {
  console.log(`${signal} 收到，开始优雅关闭...`);
  server.close(() => {
    console.log('HTTP 服务器已关闭');
    mongoose.connection.close(false, () => {
      console.log('数据库连接已关闭');
      process.exit(0);
    });
  });
};

process.on('SIGTERM', () => gracefulShutdown('SIGTERM'));
process.on('SIGINT', () => gracefulShutdown('SIGINT'));
```

## Nginx 反向代理

Nginx 作为反向代理，可以提供负载均衡、SSL 终止和静态文件服务。

### 基础配置

```nginx
# /etc/nginx/sites-available/myapp
server {
    listen 80;
    server_name myapp.com;

    location / {
        proxy_pass http://localhost:3000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
    }

    location /static {
        alias /var/www/myapp/public;
        expires 30d;
    }
}
```

### HTTPS 配置

```nginx
server {
    listen 443 ssl http2;
    server_name myapp.com;

    ssl_certificate /etc/letsencrypt/live/myapp.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/myapp.com/privkey.pem;

    location / {
        proxy_pass http://localhost:3000;
        proxy_set_header Host $host;
        proxy_set_header X-Forwarded-Proto https;
    }
}

# HTTP 重定向到 HTTPS
server {
    listen 80;
    server_name myapp.com;
    return 301 https://$server_name$request_uri;
}
```

## Docker 容器化部署

### Dockerfile

```dockerfile
# 使用 Node 官方镜像
FROM node:20-alpine

# 设置工作目录
WORKDIR /app

# 复制依赖文件
COPY package*.json ./

# 安装生产依赖
RUN npm ci --only=production

# 复制应用代码
COPY . .

# 暴露端口
EXPOSE 3000

# 健康检查
HEALTHCHECK --interval=30s --timeout=3s --start-period=5s --retries=3 \
  CMD node healthcheck.js

# 启动应用
USER node
CMD ["node", "server.js"]
```

### Docker Compose

```yaml
# docker-compose.yml
version: '3.8'

services:
  app:
    build: .
    container_name: express-app
    restart: unless-stopped
    ports:
      - "3000:3000"
    environment:
      - NODE_ENV=production
      - DATABASE_URL=mongodb://mongo:27017/myapp
      - JWT_SECRET=${JWT_SECRET}
    depends_on:
      - mongo
      - redis
    networks:
      - app-network

  mongo:
    image: mongo:7
    container_name: express-mongo
    restart: unless-stopped
    volumes:
      - mongo-data:/data/db
    networks:
      - app-network

  redis:
    image: redis:7-alpine
    container_name: express-redis
    restart: unless-stopped
    networks:
      - app-network

  nginx:
    image: nginx:alpine
    container_name: express-nginx
    restart: unless-stopped
    ports:
      - "80:80"
      - "443:443"
    volumes:
      - ./nginx.conf:/etc/nginx/conf.d/default.conf
      - ./ssl:/etc/nginx/ssl
    depends_on:
      - app
    networks:
      - app-network

volumes:
  mongo-data:

networks:
  app-network:
    driver: bridge
```

### 构建与运行

## CI/CD 流水线示例

```yaml
# .github/workflows/deploy.yml
name: Deploy

on:
  push:
    branches: [main]

jobs:
  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4
        with:
          node-version: '20'
      - run: npm ci
      - run: npm test

  deploy:
    needs: test
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - name: Deploy to server
        run: |
          ssh ${{ secrets.SSH_USER }}@${{ secrets.SSH_HOST }} '
            cd /opt/myapp &&
            git pull origin main &&
            docker-compose down &&
            docker-compose up --build -d
          '
```

## 部署清单

生产环境上线前的检查项：

- [ ] 环境变量已正确配置（非硬编码）
- [ ] 数据库连接使用连接池
- [ ] 敏感路由已添加认证中间件
- [ ] 错误处理中间件已挂载
- [ ] 日志已配置（非 console.log）
- [ ] 进程崩溃有自动重启机制（PM2 / Docker）
- [ ] HTTPS 已启用
- [ ] 静态资源使用 CDN 或缓存
